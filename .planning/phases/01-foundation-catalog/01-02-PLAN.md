<task type="auto">
  <name>Task 2: ProductCard component</name>
  <files>app/components/catalog/ProductCard.tsx</files>
  <action>
Create product card component:

1. Use 'use client' directive (has quantity state)
2. Import Product type from '@/lib/types'
3. Props: product: Product
4. State: quantity (number, default 1, range 1-99)
5. Display:
   - Product image (use placeholder or /images/products/[name].jpg pattern)
   - Product name
   - Price with $ symbol (e.g., "$350")
   - Category label (capitalized)
   - If description contains ABV%, extract and display
6. Quantity selector:
   - Minus button: decrement, min 1
   - Display current quantity
   - Plus button: increment, max 99
7. "Agregar al carrito" button:
   - Shows selected quantity
   - On click: logs to console (Phase 2 connects to cart)
   - Visual feedback on click
8. Styling per design tokens (gray-900, etc.)

```typescript
'use client'

import { useState } from 'react'
import type { Product } from '@/lib/types'

interface ProductCardProps {
  product: Product
}

export function ProductCard({ product }: ProductCardProps) {
  const [quantity, setQuantity] = useState(1)
  const [added, setAdded] = useState(false)

  const decrement = () => setQuantity(q => Math.max(1, q - 1))
  const increment = () => setQuantity(q => Math.min(99, q + 1))

  const handleAddToCart = () => {
    // Phase 2: connect to cart context
    console.log(`Adding ${quantity} × ${product.name} to cart`)
    setAdded(true)
    setTimeout(() => setAdded(false), 2000)
  }

  // Extract ABV from description if present
  const abvMatch = product.description?.match(/(\d+%)/)
  const abv = abvMatch ? abvMatch[1] : null

  return (
    <article className="bg-white border border-gray-200 rounded p-4">
      <div className="aspect-square bg-gray-100 rounded mb-4 overflow-hidden">
        <img 
          src={product.image} 
          alt={product.name}
          className="w-full h-full object-cover"
        />
      </div>
      
      <div className="space-y-2">
        <span className="text-xs uppercase tracking-wide text-gray-500">
          {product.category}
        </span>
        
        <h3 className="font-playfair font-bold text-gray-900">
          {product.name}
        </h3>
        
        <p className="text-lg font-medium text-gray-900">
          ${product.price}
        </p>
        
        {abv && (
          <p className="text-sm text-gray-600">
            {abv} alcohol
          </p>
        )}
        
        {/* Quantity selector */}
        <div className="flex items-center gap-2 py-2">
          <button
            onClick={decrement}
            className="w-10 h-10 flex items-center justify-center border border-gray-300 rounded hover:bg-gray-100"
            aria-label="Decrease quantity"
          >
            −
          </button>
          
          <span className="w-8 text-center font-medium">
            {quantity}
          </span>
          
          <button
            onClick={increment}
            className="w-10 h-10 flex items-center justify-center border border-gray-300 rounded hover:bg-gray-100"
            aria-label="Increase quantity"
          >
            +
          </button>
        </div>
        
        <button
          onClick={handleAddToCart}
          className="w-full bg-gray-900 text-white py-3 rounded font-medium hover:bg-gray-800 transition-colors"
        >
          {added 
            ? `✓ Agregado (${quantity})` 
            : `Agregar al carrito (${quantity})`
          }
        </button>
      </div>
    </article>
  )
}
```
